Use of 'var'


    function () {
        var a = 1,
            b = 2;

        function () {
            var a = 2;
            b = 3;
        }

        console.log(a + b);
    }

Order


    function () {
        console.log(1);
        setTimeout(function () {
            console.log(2);
        }, 2000);
        setTimeout(function () {
            console.log(3)
        }, 1000);
        console.log(4);
    }

Describe the following code


    var a = (function () {
        var arr = [1, 2, 3, 4];

        return function () {
            var res = 1;
            arr.forEach(function (num) {
                res = res * num;
            });
            return res;
        };
    })();

    var b = a();

    console.log(a, b);
